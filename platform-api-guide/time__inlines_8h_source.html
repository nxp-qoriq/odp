61 static inline uint64_t _odp_time_hw_to_ns(odp_time_t time)
62 {
63  uint64_t nsec;
64  uint64_t freq_hz = _odp_time_glob.hw_freq_hz;
65  uint64_t count = time.count;
66  uint64_t sec = 0;
67 
68  if (count >= freq_hz) {
69  sec = count / freq_hz;
70  count = count - sec * freq_hz;
71  }
72 
73  nsec = (_ODP_TIME_GIGA_HZ * count) / freq_hz;
74 
75  return (sec * _ODP_TIME_GIGA_HZ) + nsec;
76 }

78 static inline uint64_t _odp_time_convert_to_ns(odp_time_t time)
79 {
80  if (_odp_time_glob.use_hw)
81  return _odp_time_hw_to_ns(time);
82 
83  return time.nsec;
84 }

154 _ODP_INLINE int odp_time_cmp(odp_time_t t2, odp_time_t t1)
155 {
156  if (odp_likely(t2.u64 > t1.u64))
157  return 1;
158 
159  if (t2.u64 < t1.u64)
160  return -1;
161 
162  return 0;
