191 _ODP_INLINE void *odp_packet_l2_ptr(odp_packet_t pkt, uint32_t *len)
192 {
193  uint32_t offset = odp_packet_l2_offset(pkt);
194  uint32_t seg_len = odp_packet_seg_len(pkt);
195  uint8_t *data = (uint8_t *)odp_packet_data(pkt);
196 
197  if (odp_unlikely(offset >= seg_len)) {
198  void *pkt_hdr = (void *)pkt;
199 
200  return _odp_packet_map(pkt_hdr, offset, len, NULL);
201  }
202 
203  if (len)
204  *len = seg_len - offset;
205 
206  return data + offset;
207 }
