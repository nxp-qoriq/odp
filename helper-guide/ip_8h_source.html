109 static inline int odph_ipv4_csum(odp_packet_t pkt,
110  uint32_t offset,
111  odph_ipv4hdr_t *ip,
112  odp_u16sum_t *chksum)
113 {
114  unsigned nleft = ODPH_IPV4HDR_IHL(ip->ver_ihl) * 4;
115  uint16_t buf[nleft / 2];
116  int res;
117 
118  if (odp_unlikely(nleft < sizeof(*ip)))
119  return -1;
120  ip->chksum = 0;
121  memcpy(buf, ip, sizeof(*ip));
122  res = odp_packet_copy_to_mem(pkt, offset + sizeof(*ip),
123  nleft - sizeof(*ip),
124  buf + sizeof(*ip) / 2);
125  if (odp_unlikely(res < 0))
126  return res;
127 
128  *chksum = ~odp_chksum_ones_comp16(buf, nleft);
129 
130  return 0;
131 }

140 static inline int odph_ipv4_csum_valid(odp_packet_t pkt)
141 {
142  uint32_t offset;
143  int res;
144  odph_ipv4hdr_t ip;
145  odp_u16sum_t chksum, cur_chksum;
146 
147  offset = odp_packet_l3_offset(pkt);
148  if (offset == ODP_PACKET_OFFSET_INVALID)
149  return 0;
150 
151  res = odp_packet_copy_to_mem(pkt, offset, sizeof(odph_ipv4hdr_t), &ip);
152  if (odp_unlikely(res < 0))
153  return 0;
154 
155  chksum = ip.chksum;
156 
157  res = odph_ipv4_csum(pkt, offset, &ip, &cur_chksum);
158  if (odp_unlikely(res < 0))
159  return 0;
160 
161  return (cur_chksum == chksum) ? 1 : 0;
162 }

172 static inline int odph_ipv4_csum_update(odp_packet_t pkt)
173 {

175  odph_ipv4hdr_t ip;
176  odp_u16sum_t chksum;
177  int res;
178 
179  offset = odp_packet_l3_offset(pkt);
180  if (offset == ODP_PACKET_OFFSET_INVALID)
181  return -1;
182 
183  res = odp_packet_copy_to_mem(pkt, offset, sizeof(ip), &ip);
184  if (odp_unlikely(res < 0))
185  return res;
186 
187  res = odph_ipv4_csum(pkt, offset, &ip, &chksum);
188  if (odp_unlikely(res < 0))
189  return res;
190

192  offset + ODPH_IPV4HDR_CSUM_OFFSET,
